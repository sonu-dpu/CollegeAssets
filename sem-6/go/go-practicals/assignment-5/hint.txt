1.	Write a program in GO language to demonstrate working of slices (like append, remove, copy etc.)
Sol:-   package main
import "fmt"
func main() {

	var n int

	// Taking size of slice from user
	fmt.Print("Enter number of elements: ")
	fmt.Scan(&n)

	// Creating slice
	nums := make([]int, n)

	// Taking elements from user
	fmt.Println("Enter elements:")
	for i := 0; i < n; i++ {
		fmt.Scan(&nums[i])
	}

	fmt.Println("\nOriginal slice:", nums)

	// 1. Append operation
	var newElement int
	fmt.Print("\nEnter element to append: ")
	fmt.Scan(&newElement)

	nums = append(nums, newElement)
	fmt.Println("After append:", nums)

	// 2. Remove element
	var index int
	fmt.Print("\nEnter index to remove element: ")
	fmt.Scan(&index)

	if index >= 0 && index < len(nums) {
		nums = append(nums[:index], nums[index+1:]...)
		fmt.Println("After removing element:", nums)
	} else {
		fmt.Println("Invalid index")
	}

	// 3. Copy slice
	copySlice := make([]int, len(nums))
	copy(copySlice, nums)
	fmt.Println("\nCopied slice:", copySlice)

}

3.	Write a program in GO language to accept one matrix and display its transpose.
Sol:-  package main
import "fmt"
func main() {

	var rows, cols int

	// Accept matrix size
	fmt.Print("Enter number of rows: ")
	fmt.Scan(&rows)

	fmt.Print("Enter number of columns: ")
	fmt.Scan(&cols)

	// Create matrix
	matrix := make([][]int, rows)
	for i := 0; i < rows; i++ {
		matrix[i] = make([]int, cols)
	}

	// Accept matrix elements
	fmt.Println("Enter matrix elements:")
	for i := 0; i < rows; i++ {
		for j := 0; j < cols; j++ {
			fmt.Scan(&matrix[i][j])
		}
	}

	// Display original matrix
	fmt.Println("\nOriginal Matrix:")
	for i := 0; i < rows; i++ {
		for j := 0; j < cols; j++ {
			fmt.Print(matrix[i][j], " ")
		}
		fmt.Println()
	}

	// Create transpose matrix
	transpose := make([][]int, cols)
	for i := 0; i < cols; i++ {
		transpose[i] = make([]int, rows)
	}

	// Compute transpose
	for i := 0; i < rows; i++ {
		for j := 0; j < cols; j++ {
			transpose[j][i] = matrix[i][j]
		}
	}

	// Display transpose
	fmt.Println("\nTranspose Matrix:")
	for i := 0; i < cols; i++ {
		for j := 0; j < rows; j++ {
			fmt.Print(transpose[i][j], " ")
		}
		fmt.Println()
	}
}
4.	Write a program in GO language to accept two matrices and display its multiplication.
Sol:- package main
import "fmt"
func main() {
	var r1, c1, r2, c2 int

	// Accept size of first matrix
	fmt.Print("Enter rows and columns of first matrix: ")
	fmt.Scan(&r1, &c1)

	// Accept size of second matrix
	fmt.Print("Enter rows and columns of second matrix: ")
	fmt.Scan(&r2, &c2)

	// Check multiplication condition
	if c1 != r2 {
		fmt.Println("Matrix multiplication not possible!")
		fmt.Println("Columns of first matrix must be equal to rows of second matrix.")
		return
	}

	// Declare matrices
	mat1 := make([][]int, r1)
	mat2 := make([][]int, r2)
	result := make([][]int, r1)

	for i := range mat1 {
		mat1[i] = make([]int, c1)
	}

	for i := range mat2 {
		mat2[i] = make([]int, c2)
	}

	for i := range result {
		result[i] = make([]int, c2)
	}

	// Input first matrix
	fmt.Println("Enter elements of first matrix:")
	for i := 0; i < r1; i++ {
		for j := 0; j < c1; j++ {
			fmt.Scan(&mat1[i][j])
		}
	}

	// Input second matrix
	fmt.Println("Enter elements of second matrix:")
	for i := 0; i < r2; i++ {
		for j := 0; j < c2; j++ {
			fmt.Scan(&mat2[i][j])
		}
	}

	// Matrix multiplication logic
	for i := 0; i < r1; i++ {
		for j := 0; j < c2; j++ {
			for k := 0; k < c1; k++ {
				result[i][j] += mat1[i][k] * mat2[k][j]
			}
		}
	}

	// Display result
	fmt.Println("Resultant Matrix after multiplication:")
	for i := 0; i < r1; i++ {
		for j := 0; j < c2; j++ {
			fmt.Printf("%d ", result[i][j])
		}
		fmt.Println()
	}
}
5.	Write a program in GO language to accept n records of employee information (eno,ename,salary) and display records of employees having minimum salary.
Sol:-   package main
import "fmt”
// Define structure
type Employee struct {
	eno    int
	ename  string
	salary float64
}
func main() {
	var n int

	// Accept number of employees
	fmt.Print("Enter number of employees: ")
	fmt.Scan(&n)

	// Create slice of employees
	emp := make([]Employee, n)

	// Accept employee details
	fmt.Println("Enter employee details:")
	for i := 0; i < n; i++ {
		fmt.Printf("\nEmployee %d\n", i+1)
		fmt.Print("Enter Employee Number: ")
		fmt.Scan(&emp[i].eno)

		fmt.Print("Enter Employee Name: ")
		fmt.Scan(&emp[i].ename)

		fmt.Print("Enter Salary: ")
		fmt.Scan(&emp[i].salary)
	}

	// Find minimum salary
	minSalary := emp[0].salary
	for i := 1; i < n; i++ {
		if emp[i].salary < minSalary {
			minSalary = emp[i].salary
		}
	}

	// Display employees with minimum salary
	fmt.Println("\nEmployees having minimum salary:")
	for i := 0; i < n; i++ {
		if emp[i].salary == minSalary {
			fmt.Println("Employee No:", emp[i].eno)
			fmt.Println("Employee Name:", emp[i].ename)
			fmt.Println("Salary:", emp[i].salary)
			fmt.Println()
		}
	}
}
6.	Write a program in GO language to sort array elements in ascending order.
Sol:-   package main

import "fmt"

func main() {
	var n int

	// Accept size of array
	fmt.Print("Enter number of elements: ")
	fmt.Scan(&n)

	var arr [100]int

	// Accept array elements
	fmt.Println("Enter array elements:")
	for i := 0; i < n; i++ {
		fmt.Scan(&arr[i])
	}

	// Sorting logic (Bubble Sort)
	for i := 0; i < n-1; i++ {
		for j := 0; j < n-i-1; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j] // swap
			}
		}
	}

	// Display sorted array
	fmt.Println("Array elements in ascending order:")
	for i := 0; i < n; i++ {
		fmt.Print(arr[i], " ")
	}
}

8.	Write a program in GO language to accept n student details like roll_no, stud_name, mark1,mark2, mark3. Calculate the total and average of marks using structure.
Sol:- package main

import "fmt"

// Define Student structure
type Student struct {
	rollNo   int
	studName string
	mark1    float64
	mark2    float64
	mark3    float64
	total    float64
	average  float64
}

func main() {
	var n int

	// Accept number of students
	fmt.Print("Enter number of students: ")
	fmt.Scan(&n)

	// Create slice of students
	students := make([]Student, n)

	// Accept student details and calculate total & average
	for i := 0; i < n; i++ {
		fmt.Printf("\nEnter details of Student %d\n", i+1)

		fmt.Print("Enter Roll Number: ")
		fmt.Scan(&students[i].rollNo)

		fmt.Print("Enter Student Name: ")
		fmt.Scan(&students[i].studName)

		fmt.Print("Enter Mark1, Mark2, Mark3: ")
		fmt.Scan(&students[i].mark1, &students[i].mark2, &students[i].mark3)

		// Calculate total
		students[i].total = students[i].mark1 +
			students[i].mark2 +
			students[i].mark3

		// Calculate average
		students[i].average = students[i].total / 3
	}

	// Display student details
	fmt.Println("\nStudent Details with Total and Average:")
	for i := 0; i < n; i++ {
		fmt.Println("\nRoll No:", students[i].rollNo)
		fmt.Println("Name:", students[i].studName)
		fmt.Println("Total Marks:", students[i].total)
		fmt.Println("Average Marks:", students[i].average)
	}
}
